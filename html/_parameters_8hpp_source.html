00272     //The static_cast<T*>(0) in get is just a tricky way to disambiguate the call.

00282 const T Parameters::get_impl(const std::string& key, T*)
00283 {
00284 
00285     Map::iterator i = params_.find(key);
00286     if (i == params_.end()) {
00287         throw InvalidParameterName("Parameter does not exist: " + key);
00288     }
00289     try {
00290         return boost::lexical_cast<T>(boost::get<std::string>(i->second));
00291     } catch (boost::bad_lexical_cast&) {
00292         throw InvalidParameterType(
00293             "Request for parameter " + key + " wrong conversation.");
00294     } catch (boost::bad_get&) {

00296             "Request for parameter " + key + " expects wrong type.");
