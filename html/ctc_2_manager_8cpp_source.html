00051 void Manager::processRequest(libpipe::Request& req)
00052 {
00053 if (req.is (Request::UPDATE)) {
00054             if (!algorithm_) {
00055                         throw RequestException(
00056                         "Cannot process request. No algorithm setup available.");
00057                     }
00058                     typedef FilterSet::iterator MSI;
00059                     // iterate over all sources
00060                     for (MSI i = sources_.begin(); i != sources_.end(); ++i) {
00061             try {
00062                 (*i)->processRequest(req);
00063             } catch (RequestException& e) {
00064                 throw;
00065             }
00066         }
00067     }
00068     try {
00069         algorithm_->processRequest(req);
00070     } catch (std::exception& e) {
00071         std::string str(e.what());
00072         throw RequestException(
00073         "ModificationTimeManager: Cannot process request: algorithm execution caused exception: "
00074         + str);
00075     } catch (...) {
00076         throw RequestException(
00077         "ModificationTimeManager: Cannot process request: algorithm execution caused exception.");
00078     }
00079     if (req.is(Request::DELETE)) {
00080         this->disconnect();
00081     }
00082 }
