00075 bool Parameters::validate() const
00076 {
00077     // The set function makes sure that all parameters that are set are
00078     // valid. Hence, to validate, we only need to check if all required
00079     // parameters have been set.
00080     std::vector<std::string>::const_iterator i;
00081     for (i = requiredParams_.begin(); i != requiredParams_.end(); ++i) {
00082         if (!params_.count(*i)) {
00083             return false;
00084         }
00085     }
00086     return true;
00087 }
