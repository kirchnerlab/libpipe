00053         std::string tempStr;
00054         if (Pipeline::getTraceFlag()) {
00055             if (tempReq.getType() == libpipe::Request::UPDATE) {
00056                 tempStr = "Starting Update Request";
00057             } else
00058                 if (tempReq.getType() == libpipe::Request::DELETE) {
00059                     tempStr = "Starting Delete Request";

00061         }
00062 
00063         LIBPIPE_PIPELINE_TRACE(tempStr);
00064 
00065         pipelineQueue_.front()->getManager()->processRequest(tempReq);
00066 
00067         pipelineQueue_.pop();
00068         requestQueue_.pop();
00069     }
00070 
00071 }
