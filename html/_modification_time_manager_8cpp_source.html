00043 void ModificationTimeManager::processRequest(libpipe::Request& req)
00044 {
00045     // make sure we have been set up correctly
00046     if (!algorithm_) {
00047         throw RequestException(
00048             "Cannot process request. No algorithm setup available.");
00049     }
00050     // check if we have ever run at all
00051     bool needUpdate = algorithm_->needUpdate();
00052     // iterate over all sources and check their modification time
00053     typedef FilterSet::iterator FSI;
00054     for (FSI i = sources_.begin(); i != sources_.end(); ++i) {
00055         try {
00056             (*i)->processRequest(req);
00057         } catch (RequestException& e) {
00058             throw;
00059         }
00060         // check the modification time and see if we need to update
00061         if (!needUpdate && algorithm_->getMTime()
00062                 <= (*i)->getAlgorithm()->getMTime()) {
00063             needUpdate = true;
00064         }
00065     }
00066     // update, if necessary
00067     if (needUpdate) {
00068         try {
00069             algorithm_->processRequest(req);
00070         } catch (std::exception& e) {
00071             std::string str(e.what());
00072             throw RequestException(
00073                 "ModificationTimeManager.cpp: Cannot process request: algorithm execution caused exception: "
00074                         + str);
00075         } catch (...) {
00076             throw RequestException(
00077                 "ModificationTimeManager.cpp: Cannot process request: algorithm execution caused exception.");
00078         }
00079     }
00080 }
