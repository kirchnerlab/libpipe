00099     for (std::list<FilterDescription>::const_iterator it = filterList.begin();

00156 {
00157     std::priority_queue<PipelineDescription, std::vector<PipelineDescription>,
00158             PipelineDescriptionLess> queue = configuration_->getLibpipePipe();
00159 
00160     while (!queue.empty()) {
00161         PipelineDescription temp = queue.top();
00162         queue.pop();
00163 
00164         if (temp.requestType == "UPDATE") {
00165             Request tempReq(libpipe::Request::UPDATE);
00166             pipeline_.setTraceFlag(temp.makeTrace);
00167             pipeline_.push(tempReq, this->getFilter(temp.filterName));
00168         } else if (temp.requestType == "DELETE") {
00169             Request tempReq(libpipe::Request::DELETE);
